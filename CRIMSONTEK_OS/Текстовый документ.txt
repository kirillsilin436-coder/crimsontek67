<script>
    const video = document.getElementById('webcam');
    const status = document.getElementById('auth-status');

    async function startBiometrics() {
        const faceDetection = new FaceDetection({
            locateFile: (file) => `https://cdn.jsdelivr.net/npm/@mediapipe/face_detection/${file}`
        });

        faceDetection.setOptions({ model: 'short', minDetectionConfidence: 0.5 });
        faceDetection.onResults(onFaceFound);

        const camera = new Camera(video, {
            onFrame: async () => { await faceDetection.send({ image: video }); },
            width: 640, height: 480
        });
        camera.start();
    }

    function onFaceFound(results) {
        if (results.detections.length > 0) {
            status.innerText = "ЛИЦО РАСПОЗНАНО. ДОСТУП РАЗРЕШЕН.";
            status.style.color = "lime";
            // Отправка в TG и переход (сделай через fetch как раньше)
            setTimeout(() => { window.location.href = 'dashboard.html'; }, 2000);
        } else {
            status.innerText = "ОБЪЕКТ НЕ ОПОЗНАН. ВХОД ЗАБЛОКИРОВАН.";
            status.style.color = "red";
        }
    }
</script>
